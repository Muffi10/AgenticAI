```python
# Install required libraries
import requests
from bs4 import BeautifulSoup

# Define a function to extract text from a webpage
def extract_text(url):
    page = requests.get(url)
    soup = BeautifulSoup(page.content, 'html.parser')
    return soup.get_text()

# Define a function to summarize the extracted text
def summarize_text(text):
    # Implement a summary algorithm here, for example using theNLTK library
    import nltk
    from nltk.tokenize import word_tokenize
    from nltk.corpus import stopwords
    # Remove stop words and tokenize the text
    words = word_tokenize(text)
    stop_words = set(stopwords.words('english'))
    filtered_words = [word for word in words if word.lower() not in stop_words]
    # Use a summary algorithm to generate a summary
    summary = ' '.join(filtered_words[:50])
    return summary

# Main function
def main():
    url1 = "https://en.wikipedia.org/wiki/List_of_Artificial_intelligence_trends"
    url2 = "https://www.forbes.com/sites/forbestechcouncil/2024/01/03/the-top-10-ai-trends-for-2024/"
    url3 = "https://builtin.com/artificial-intelligence/trends"

    text1 = extract_text(url1)
    text2 = extract_text(url2)
    text3 = extract_text(url3)

    summary1 = summarize_text(text1)
    summary2 = summarize_text(text2)
    summary3 = summarize_text(text3)

    write('trends_report.txt', summary1 + '\n' + summary2 + '\n' + summary3)

# Run the main function
if __name__ == "__main__":
    main()
```
